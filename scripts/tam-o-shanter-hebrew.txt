כַּאֲשֶׁר הַתַּגְרָנִים יוֹצְאִים אֶת הָרְחוֹבוֹת,
וּשְׁכֵנִים, שְׁכֵנִים צְמֵאִים נִפְגָּשִׁים;
כְּשֶׁיְּמֵי הַשּׁוּק מַגִּיעִים לְתֻמָּם,
וְהַקָּהָל פּוֹנִים לַבָּתִּים בְּדַרְכָּם,
כְּשֶׁאָנוּ סוֹבְאִים שֵׁכָר שְׂעוֹרִים,
מִשְׁתַּכְּרִים וְנֶעֱטָפִים שִׂמְחָה יְתֵרָה,
אֵין חוֹשְׁבִים עַל הַמִּילִין שֶׁל דַּרְכֵי סְקוֹט,
מִקְוֵי מַיִם, מַעֲבָרִים, מִכְשׁוֹלִים וּבִצּוֹת,
הַפְּרוּשִׂים בֵּינֵינוּ וְעַד לַבַּיִת,
שָׁם יוֹשֶׁבֶת גְּבִרְתֵּנוּ זְעוּפָה וקוֹדֶרֶת,
אֶת גְּבִינֶיהָ מְכַוֶּצֶת כִּסְעָרָה נֶאֱסֶפֶת,
זַעְמָהּ בּוֹעֵר, לֹא נִגְרַע בּוֹ כַּזַּיִת.

זוֹ הָאֱמֶת שֶׁל טָם אוֹ'שַׁנְטֶר הַצַּדִּיק
שֶׁבְּלַיְלָה אֶחָד מֵאֵיר רָכַב,
(אֵיר הַיְשָׁנָה, שֶׁאֵין בִּלְתָּהּ
בְּצַדִּיקוּת בָּחוּרֶיהָ וּבִיפִי עַלְמוֹתֶיהָ.)

הוֹ טָם, לוּ רַק הֶחְכַּמְתָּ
לְקַבֵּל אֶת עֲצָתָהּ שֶׁל קֵיט אִשְׁתְּךָ!
הֵיטֵב הִבְהִירָה לְךָ הַבַּטְלָן,
הַמְבֻשָּׂם, הַיָּהִיר, הַבַּרְבְּרָן, הָרַבְרְבָן,
אֲשֶׁר מִּנּוֹבֶמְבֶּר עַד אוֹקְטוֹבֶּר,
לֹא יִמְצָאוּךָ צָלוּל בְּאַף יוֹם שׁוּק;
אֲשֶׁר בְּעֵת קָצִיר עִם הַטּוֹחֵן יָשַׁבְתָּ,
כָּל עוֹד נוֹתְרָה בְּכִיסְךָ הַפְּרוּטָה,
אַתָּה וְהַמְפַרְזֵל שְׁתִיתֶם לְשָׁכְרָה,
לְכָל סוּס שֶׁהוּא הִרְכִּיב לוֹ פַּרְסָה;
שֶׁבְּבֵית הַכְּנֵסִיָּה, בְּשַׁבָּתוֹן יוֹם רִאשׁוֹן,
שָׁתִיתָ עַד יוֹם שֵׁנִי עִם ג'ין קִירְקְטוֹן.
קֵיט נִבְּאָה שֶׁבְּמֻקְדָּם אוֹ בִּמְאֻחָר,
יִמְצָאוּךָ עָמֹק בַּדּוּן טָבוּעַ,
אוֹ יִלְכְּדוּךָ מְכַשְּׁפִים בַּעֲלָטָה,
בִּכְנֵסִיַּת אַלוֹוֵי שֶׁהָרוּחוֹת רַדְּפוּהָ.

אָה, גְּבִירוֹת כְּבוּדוֹת, אֲנִי בוֹכֶה,
כְּשֶׁחוֹשֵׁב אֲנִי כַּמָּה עֵצוֹת מְתוּקוֹת,
כַּמָּה הַצָּעוֹת מְדֻיָּקוֹת וּנְבוֹנוֹת
מֵאֶת אִשְׁתּוֹ הַבַּעַל בְּבוּז דּוֹחֶה!

נָשׁוּב אֶל סִפּוּרֵנוּ:– בְּלֵיל שׁוּק אֶחָד,
יָשַׁב טָם בְּצוּרָה נוֹחָה,
לְצַד אָח בּוֹעֶרֶת, מְפַזֶּזֶת,
עִם מַשְׁקֶה בִּירָה שְׁמֵימִית מַקְצִיפָה,
וּלְצִדּוֹ קוֹבְּלֶר ג'וֹנִי הַסַּנְדְּלָר,
יְדִידוֹ הֶעָתִיק שֶׁגְּרוֹנוֹ נִחָר;
טָם אָהַב אוֹתוֹ מַמָּשׁ כְּאָח,
שָׁבוּעוֹת בְּגִלּוּפִין הֵם בִּלּוּ יַחַד.
הַלַּיְלָה עָבַר בְּשִׁירָה וּבַהֲמֻלָּה,
וְעִם כָּל מַשְׁקֶה הַטַּעַם הֻשְׁבָּח;
בַּעֲלַת הַבַּיִת וְטָם עָסְקוּ בִּנְדִיבוּת,
בְּמַעֲשֵׂי חֶסֶד חֲשָׁאִיִּים שֶׁל מְתִיקוּת;
הַסַּנְדְּלָר סִפֵּר מִמַּעֲשִׂיּוֹתָיו הַמְּשֻׁנּוֹת;
בַּעַל הַבַּיִת נַעֲנָה בִּצְחוֹקוֹ בְּרָצוֹן:
בַּחוּץ הַסְּעָרָה תִּרְחֹשׁ וְתִרְעֹם,
טָם לֹא הִתְרַגֵּשׁ מִמֶּנָּה בִּמְאוּם.

שִׁגָּעוֹן לִרְאוֹת גֶּבֶר כֹּה שָׂמֵחַ,
גַּם אִם בְּשֵׁכָר הוּא עַצְמוֹ טוֹבֵעַ.
כִּדְבוֹרִים שֶׁעָפוֹת עֲמוּסוֹת מַטְעָמִים,
הַדַּקּוֹת עָפוּ עָבְרוּ לָהֶם בִּנְעִימִים:
הַמֶּלֶךְ מְבֹרָךְ הוּא, אֲבָל טָם הַמְּהֻלָּל
גָּבַר עַל כָּל תַּחֲלוּאֵי הַחַיִּים.

אֲבָל תַּעֲנוּגוֹת כְּפִרְחֵי פְּרָגִים מִתְפַּזְּרִים:
אַךְ תֹּאחַז בְּפֶרַח וְעָלָיו נוֹשְׁרִים;
אוֹ כְּרֶדֶת הַשֶּׁלֶג עַל הַנְּחָלִים,
לָבָן לְרֶגַע – וְנָמֵס לְעוֹלָמִים,
אוֹ כְּקַרְנֵי אוֹרֹת הַצָּפוֹן שֶׁרוֹאִים,
אִם אַךְ תַּצְבִּיעַ וְהֵם נְמוֹגִים;
אוֹ כְּצוּרָה מֻפְלָאָה שֶׁל קֶשֶׁת בֶּעָנָן,
הַנֶּעֱלֶמֶת מִיָּד עִם רוּחַ רַעֲנָן.
אֵין אָדָם רוֹצֵעַ אֶת הַזְּמַן אוֹ הַגֵּאוּת שֶׁל הַיָּם,
הִגִּיעָה הָעֵת לִרְכִיבָתוֹ שֶׁל טָם:
שָׁעָה שֶׁל שִׂיא כִּפַּת שְׁחֹר הַלַּיִל,
בְּעֵת זוֹ הַקּוֹדֶרֶת הוּא מְטַפֵּס עַל בְּהֶמְתּוֹ
וּבְלַיְלָה כְּזֶה הוּא יוֹצֵא לְדַרְכּוֹ
כְּפִי שֶׁאַף חוֹטֵא אֻמְלָל לֹא אִתְּרַע מַזָּלוֹ.

הָרוּחַ נָשַׁב כְּאִלּוּ בַּפַּעַם הָאַחֲרוֹנָה;
גְּשָׁמִים חָבְטוּ בְּקוֹל בָּאֲדָמָה;
בִּרְקֵי בָזָק נִבְלְעוּ בַּחֲשֵׁכָה,
הָרַעַם הִרְעִים בִּשְׁאָגָה עֲמוּמָה:
אוֹתוֹ לַיְלָה, יָבִין כָּל יֶלֶד קָטָן,
הָיָה עִסְקוֹ הַמֻּבְהָק שֶׁל הַשָּׂטָן.

יָשׁוּב הֵיטֵב עַל מֶג הָאֲפֹרָה.
טוֹבָה מִמֶּנָּה לֹא הִשִּׂיאָה רַגְלָהּ,
טָם אָץ דֶּרֶךְ בֹּץ וָרֶפֶשׁ,
מְזַלְזֵל בָּרוּחַ, בַּגֶּשֶׁם וָאֵשׁ;
אוֹחֵז בְּמִצְנַפְתּוֹ הַטּוֹבָה הַכְּחֻלָּה,
מְזַמְזֵם בִּסְקוֹטִית סוֹנֵטָה,
סוֹקֵר סְבִיבָתוֹ בִּזְהִירוּת שְׁקוּלָה,
שֶׁמָּא שֵׁדִים יָכִינוּ הַפְתָּעָה:
כְּנֵסִיַּת אַלוֹוֵי קְרֵבָה וְהוֹלֶכֶת,
שָׁם שֵׁדִים וְלִילִיּוֹת צוֹוְחִים בַּלֵּילוֹת.

כָּעֵת לְאַחַר שֶׁצָּלַח אֶת הַנַּחַל,
בַּמָּקוֹם שָׁם בַּשֶּׁלֶג הַתַּגְרָן אָז נֶחְבַּל;
וּמֵעֵבֶר לַעֲצֵי לִבְנֶה וְסֶלַע עֲנָק,
שָׁם צַ'רְלִי הַשִּׁכּוֹר שָׁבַר הַמִּפְרָק;
וְדֶרֶךְ קוֹצִים, וּמֵעֵבֶר לַגַּלְעֵד,
שָׁם גִּלּוּ צַיָּדִים אֶת גּוּפָתוֹ שֶׁל הַיֶּלֶד;
וּמֵעַל לַבְּאֵר, לְיַד הָעֻזְרָר,
שָׁם אִמּוֹ שֶׁל מוּנְגוֹ קָשְׁרָה חֶבֶל לַצַּוָּאר.
לְפָנָיו נַחַל הַדּוּן מֵטִיל אֶת מֵימָיו;
הַסְּעָרָה הַמִּתְגַּבֶּרֶת רוֹעֶמֶת בַּיַּעַר;
הַבָּרָק מַבְזִיק מִקֹּטֶב אֶל קֹטֶב;
רַעַם מִתְגַּלְגֵּל הוֹלֵךְ וְקָרֵב:
בְּנִצְנוּץ מִבֵּין עֵצִים נֶאֱנָקִים,
כְּנֵסִיַּת אַלוֹוֵי נִדְמֵית בּוֹעֶרֶת,
קַרְנֵי אוֹר דֶּרֶךְ כָּל חָרָךְ מְצִיצִים,
מִצְטַלְצְלִים בַּעֲלִיצוּת וּמְרַקְּדִים.

וִויסְקִי ג'וֹן בַּרְלִיקוֹרְן מֵעוֹרֵר הַשְׁרָאָה!
בְּחָסוּתוֹ לֹא נָחוּשׁ בְּכָל סַכָּנָה!
עִם הַשֵּׁכָר לֹא נִירָא כָּל רַע;
עִם הַוִויסְקִי נַעֲמֹד בִּפְנֵי הַשָּׂטָן!
הַבִּירָה שָׂחֲתָה בְּרֹאשׁוֹ שֶׁל טָם,
וְעַל הַשֵּׁדִים הוּא לֹא שָׁם פְּרוּטָה.
אֲבָל מָגִי עָמְדָה, בִּפְלִיאָה כְּאוּבָה,
עַד שֶׁעָקֵב בְּצַד אֲגוּדָל בִּזְהִירוּת רַבָּה,
עַזְרָה עֹז וְהִתְקַדְּמָה אֶל הָאוֹר הַנִּרְאָה;
וְהוֹי! טָם רָאָה מַרְאֶה מַדְהִים!

מְכַשְּׁפוֹת וְיִדְּעוֹנִים יַחְדָּיו מְרַקְּדִים:
לֹא בִּקְוַדְרִיל צָרְפָתִי אָפְנָתִי,
רַק רִקּוּד סְקוֹטִי מָהִיר, אִטִּי, קוֹפְצָנִי,
הִרְעִיפוּ חַיִּים וֶעֱזוּז רַגְלִיִּים.
בְּגֻמְחַת חַלּוֹן מִזְרָחִית,
יָשַׁב אַשְׁמְדַאי בְּצוּרָה בְּהַמְתִית;
כֶּלֶב שָׁחֹר מְדֻבְלָל, קוֹדֵר וְגָדוֹל,
מַשְׁמִיעַ לָהֶם מוּסִיקָה לְמָחוֹל:
בִּצְוָחוֹת חַדּוֹת סָחַט אֶת קוֹלוֹ,
עַד שֶׁקּוֹרוֹת הַגַּג גַּם הֵם צִלְצְלוּ.
אֲרוֹנוֹת מֵתִים נִפְעֲרוּ כְּמַכְבְּשִׁים,
הִצִּיגוּ בְּנֵי-מִינָן בִּלְבוּשָׁם הַסּוֹפִי;
כָּל אֶחָד בְּמֶחֱוָה שְׂטָנִית פִּלְאִית,
אָחַז בְּנֵר בּוֹעֵר בְּיָדוֹ הַקָּרָה:
לְאוֹרָם טָם הַגִּבּוֹר
רָאָה עַל שֻׁלְחַן הַקֹּדֶשׁ,
עֲצָמוֹת רוֹצֵחַ, בִּסְבָכַת בַּרְזֶל;
שְׁנֵי תִּינוֹקוֹת אָרְכָּם כַּזֶּרֶת, לִפְנֵי הַטְּבִילָה;
גַּנָּב שֶׁהוּרַד זֶה עַתָּה מֵחֶבֶל הַתְּלִיָּה –
פִּיו פָּעוּר בְּשֶׁנֶק נְשִׁימָתוֹ הָאַחֲרוֹנָה;
חֲמִשָּׁה גַּרְזִנֵי טוֹמָהוֹק עִם דָּם חָלוּד;
חָמֵשׁ חַרְבוֹת סִימִיטָר עִם רֶצַח קָרוּשׁ;
בִּירִית בָּהּ תִּינוֹק נֶחֱנָק;
סַכִּין בָּהּ נֶחְתַּךְ גְּרוֹנוֹ שֶׁל אָב,
שֶׁבְּנוֹ שֶׁלּוֹ נִשְׁלְלוּ חַיָיו,
שֵׂעָר אָפֹר עוֹד דָּבוּק לַכִּידוֹן;
וְעוֹד וְעוֹד זְוָעָה וְחַגָּא,
שֶׁרַק בַּהֲגִיָּתָם תִּהְיֶה כְּדֵי עֲבֵרָה.
שְׁלוֹשָׁה לְשׁוֹנוֹת שֶׁל עוֹרְכֵי דִּין, מְהֻפָּכִים,
כְּמוֹ בִּגְדוֹ שֶׁל קַבְּצָן זְרוּיִים שְׁקָרִים –
שְׁלוֹשָׁה לְבָבוֹת כְּמָרִים, רְקוּבִים, שְׁחֹרִים
זְרוּקִים בְּכָל פִּנָּה מַסְרִיחִים, מַגְעִילִים.

כְּשֶׁטָּמִי הִשְׁקִיף, בְּסַקְרָנוּת תְּמֵהָה,
גָּבְרוּ בוֹ וְהָלְכוּ הָעֲלִיצוּת וְהַשִּׂמְחָה;
הַחֲלִילָן נָשַׁף חָזָק עוֹד וָעוֹד,
הָרַקְדָנִים עָפוּ מַהֵר יוֹתֵר וְיוֹתֵר,
הֵם הִסְתַּחְרְרוּ, הֵם יָשְׁבוּ, הֵם חָצוּ, הֵם הִשְׁתַּלְּבוּ,
עַד שֶׁכָּל מְכַשֵּׁפָה הִזִּיעָה וְהִצְחִינָה,
וְהִשְׁלִיכָה בְּגָדִים מְרֻפָּטִים לָרִצְפָּה,
וְרָקְדָה בִּמְיֻמָּנוּת בְּתַחְתּוֹנִיתָהּ!

עַכְשָׁו טָם, הוֹ טָם! לוּ הָיוּ אֵלֶּה נְעָרוֹת,
שְׁמַנְמַנּוֹת וַחֲזָקוֹת בְּגִילֵי הֶעָשׂוֹר!
הַתַּחְתּוֹנִיּוֹת, בִּמְקוֹם פְלָנֵל שַׁמְנוּנִי,
הָיוּ פִּשְׁתָּן צָחוֹר בַּאֲרִיגָה עִלִּית! –
מִכְנָסַי, הַזּוּג הַיָּחִיד בִּרְשׁוּתִי,
שֶׁהָיוּ פַּעַם קְטִיפָה כְּחֻלָּה מְעֻלָּה,
הָיִיתִי מֵסִיר מֵעַכּוּזִי וְנוֹתֵן
עֲבוּר מִצְמוּץ אֶחָד שֶׁל הַיָּפוֹת הַהֵן!

אַךְ מְכַשְּׁפוֹת בָּלוֹת, זְקֵנוֹת נִלְעָגוֹת,
בְּכִעוּרָן הָרַב הֵן יַנִּיקוֹ עַיִר,
עַל מַקֵּל מַטְאֲטֵא מְדַלְּגוֹת מְרַקְּדוֹת,
פֶּלֶא שֶׁלֹּא הִתְהַפְּכוּ הַמֵּעַיִם!

אַךְ טָם הֵיטֵב הִבְחִין מַהוּ מָה:
הָיְתָה שָׁם נַעֲרָה מְצוֹדֶדֶת וְעַלִּיזָה,
שֶׁגֻּיְּסָה בְּעֶצֶם בְּאוֹתוֹ הַלַּיְלָה,
זֶה זְמַן רַב שֶׁנּוֹדְעָה בְּקָרִיק בַּחוֹף;
(שֶׁיָּרְתָה לַמָּוֶת בִּבְהֵמוֹת לָרֹב,
וְהִשְׁמִידָה הַרְבֵּה סִירוֹת נָאוֹת,
וְחַמְסָה בְּלִי סוֹף תִּירָס וּשְׂעוֹרָה,
וְהִפִּילָה חִתָּתָהּ עַל כָּל הַסְּבִיבָה.)
קָאטִי סַרְק מִבַּד פֵּיזְלִי, הַקְּצָרָה קְצָרָה
שֶׁלָּבְשָׁה עוֹד כְּנַעֲרָה צְעִירָה,
גַּם אִם זְעוּמָה הִיא בְּאָרְכָּהּ,
הִיא הַטּוֹבָה בַּמֶּלְתָּחָה, וּבָהּ הִתְגָּאֲתָה,-
הָהּ! סָבָתֵךְ הַכְּבֻדָּה לוּ רַק יָדְעָה,
שֶׁהַתַּחְתּוֹנִית שֶׁקָּנְתָה לְנֶנִי הַקְּטַנָּה,
בְּכָל רְכוּשָׁהּ, שְׁתֵּי לִירוֹת סְקוֹט
תְּפָאֵר אֵי פַעַם רִקּוּד מְכַשְּׁפוֹת!

אֲבָל כָּאן הַמַּעֲשִׂיָּה תָּכוֹף וְתִקֹּד,
הַמִּלִּים שֶׁסִּפְּרוּ יַסְפִּיקוּ לֹא עוֹד;
לַשִּׁיר כֵּיצַד נֶנִי רִקְּדָה וּבָעֲטָה
(נַעֲרָה חֲזָקָה וּגְמִישָׁה הָיְתָה);
וְכֵיצַד עָמַד שָׁם טָם מְכֻשָּׁף,
וְהָגָה בָּעֹשֶׁר שֶׁזָּכוּ בּוֹ עֵינָיו;
אַף הַשָּׂטָן הַזּוֹעֵף נָע בְּתַאֲוָה,
וְהִטַּלְטֵל וְנָשַׁף בְּכָל כּוֹחוֹתָיו;
וְאָז דִּלּוּג אֶחָד, וְאֶחָד נוֹסָף,
שֶׁטָּם הִשְׁתַּגֵּעַ בְּלִי שְׁלִיטָה בְּמַעֲשָׂיו,
וְשָׁאַג: 'יָפֶה מְאוֹד, קָאטִי סַרְק!'
וּבוֹ בָּרֶגַע יָרְדָה הָעֲלָטָה;
בְּקֹשִׁי הוּא וּמֶגִּי חָזְרוּ לִתְחִיָּה,
וְהָמוֹן מֵהַגֵּיהִנּוֹם פָּרַץ כְּבָרָק.

כְּמוֹ דְּבוֹרִים שֶׁבְּכַעֲסָן מִשְׁתַּלְּחוֹת
בְּטוֹרְפִים מְאַיְּמִים הַפּוֹלְשִׁים לַכַּוָּרוֹת;
כְּמוֹ שֶׁאוֹיְבוֹ הַמַּר תּוֹקֵף אֶת הָאַרְנָב,
וּפָאק! הוּא נִמְלַט לוֹ מִתַּחַת לָאַף;
כְּמוֹ שֶׁקְּהַל הַשּׁוּק מִתְלַהֵט בִּיעָף,
כְּשֶׁקּוֹרְאִים בְּקוֹל 'תִּפְסוּ אֶת הַגַּנָּב'!
כָּךְ מֶגִּי דוֹהֶרֶת, הַמְכַשְּׁפוֹת בְּעִקְבוֹתֶיהָ,
בִּצְעָקוֹת וּבִזְעָקוֹת שֶׁלֹּא בְּדֶרֶךְ הַטֶּבַע.

אַךְ, טָם! אַךְ, טָם! עוֹד תְּקַבֵּל כִּגְמוּלְךָ!
בַּגֵּיהִנּוֹם כְּדָג הֵרִינְג יִצְלוּ אוֹתְךָ!
לַשָּׁוְא תַמְתִּין קֵיט לְשׁוּבְךָ!
בְּקָרוֹב תִּהְיֶה קֵיט אִשָּׁה אֲבֵלָה!
דָּהֲרִי, מֶג, כְּכָל שֶׁאַתְּ יְכוֹלָה,
וְהַשִּׂיגִי אוֹתָם אֶל רֹאשׁ הַגֶּשֶׁר;
הַפְנִי לָהֶם שָׁם אֶת אֲחוֹרַיִךְ,
הֵם לֹא יַחֲצוּ אֶת זְרִימַת הַמַּיִם!
אַךְ לִפְנֵי שֶׁתַּסְפִּיק אֶל הַגֶּשֶׁר,
עָלֶיהָ לְהַשְׁאִיר אֶת הַשֵּׁדָה מֵאָחוֹר;
כִּי נֶנִי, שֶׁהִתְקַדְּמָה מְאוֹד,
לָחֲצָה בְּחָזְקָה עַל מֶג הָאֲצִילָה,
וְכִוְּנָה אֶל טָם אֶת מְלוֹא זַעְמָהּ;
אַךְ עֹז רוּחָהּ שֶׁל מֶגִּי הִפְתִּיעַ אוֹתָהּ!
בְּזִנּוּק חַד חִלְצָה אֶת אֲדוֹנָהּ,
בְּהוֹתִירָהּ מֵאָחוֹר אֶת זְנָבָהּ הָאָפֹר:
הַמְכַשֵּׁפָה אָחֲזָה בְּעַכּוּזָהּ,
וְהוֹתִירָה רַק גֶּדֶם לְמֶגִּי הַמִּסְכֵּנָה.

כָּעֵת, מִי שֶׁקּוֹרֵא מַעֲשִׂיָּה זוֹ שֶׁל אֱמֶת,
כָּל גֶּבֶר וּבֵן לָאֵם, שִׂימוּ לֵב:
כְּכָל שֶׁיַּחְפֹץ לִבְּךָ בְּמַשְׁקָאוֹת,
אוֹ שֶׁתִּתְאָבֶה לַחֲצָאִיּוֹת קְצָרוֹת,
חֲשֹׁב! אַתָּה עָלוּל לְשַׁלֵּם בְּיֹקֶר:
זְכֹר אֶת סוּסָתוֹ שֶׁל טָם אוֹ'שַׁנְטֵר.
